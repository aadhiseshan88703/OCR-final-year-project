```python
import cv2

def preprocess_image(image_path):
    image = cv2.imread(image_path)
    image = cv2.resize(image, (1024, 1024))
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    denoised = cv2.GaussianBlur(gray, (5, 5), 0)
    _, thresh = cv2.threshold(denoised, 150, 255, cv2.THRESH_BINARY)
    return thresh
```

```python
from paddleocr import PaddleOCR

ocr = PaddleOCR(lang='en')

def run_ocr(image):
    result = ocr.ocr(image)
    texts = []
    boxes = []
    confidences = []
    for line in result[0]:
        box = line[0]
        text = line[1][0]
        conf = line[1][1]
        texts.append(text)
        boxes.append(box)
        confidences.append(conf)
    return texts, boxes, confidences
```

Functions:
- `run_ocr(image)` - Extract text from image

```python
import json

def postprocess(texts, boxes, confidences):
    cleaned_texts = [t.strip() for t in texts]
    data = {
        "text": cleaned_texts,
        "boxes": boxes,
        "confidence": confidences
    }
    return data

def save_json(data, path="output/result.json"):
    with open(path, "w") as f:
        json.dump(data, f, indent=4)
```

Functions:
- `postprocess(texts, boxes, confidences)` - Format output
